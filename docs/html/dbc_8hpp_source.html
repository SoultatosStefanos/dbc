171  // Decomposes a boolean expression, given its left hand side operand
172  // Makes use of the operator overloads to deduce the right hand operand and the operation.
173  class rhs_decomposer
174  {
175  public:
176  template <typename T>
177  explicit rhs_decomposer(const T& lhs)
178  {
179  ss << lhs;
180  }
181 
182  template <typename T>
183  auto operator==(const T& rhs) -> const auto&
184  {
185  ss << " == " << rhs;
186  return *this;
187  }
188 
189  template <typename T>
190  auto operator!=(const T& rhs) -> const auto&
191  {
192  ss << " != " << rhs;
193  return *this;
194  }
195 
196  template <typename T>
197  auto operator<(const T& rhs) -> const auto&
198  {
199  ss << " < " << rhs;
200  return *this;
201  }
202 
203  template <typename T>
204  auto operator>(const T& rhs) -> const auto&
205  {
206  ss << " > " << rhs;
207  return *this;
208  }
209 
210  template <typename T>
211  auto operator<=(const T& rhs) -> const auto&
212  {
213  ss << " <= " << rhs;
214  return *this;
215  }
216 
217  template <typename T>
218  auto operator>=(const T& rhs) -> const auto&
219  {
220  ss << " >= " << rhs;
221  return *this;
222  }
223 
224  // Returns an std::string decomposition of the boolean expression, of the form:
225  //"'lhs' 'operation' 'rhs'""
226  auto decomposition() const -> auto { return ss.str(); }
227 
228  private:
229  std::stringstream ss;
230  };
