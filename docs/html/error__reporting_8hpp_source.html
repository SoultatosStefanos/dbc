132  class rhs_decomposer

153  }

157  {

159  return *this;
160  }
161 
162  template <typename T>
163  auto operator>(const T& rhs) -> const auto&
164  {
165  ss << " > " << rhs;
166  return *this;
167  }
168 
169  template <typename T>
170  auto operator<=(const T& rhs) -> const auto&
171  {

174  }

181  }
